\subsection{Analyzing False Positives and False Negatives}
\label{sec:fpr_analysis}

\begin{theorem}
Let $\detectionThreshold$ be the threshold for revealing the voucher key to the server. Then, 

		
		\begin{small}
		\begin{align}
			& \cprob{}{\exists I \subseteq \nats[\numProjections], \setSize{I} \ge \threshold: \forall \projIdx \in I, 
				\projFn[\projIdx](\genericMetricSpaceElmt) = \projFn[\projIdx](\genericMetricSpaceElmtAlt)}{\distanceNotation[\distance][\genericMetricSpaceElmt][\genericMetricSpaceElmtAlt] > 
			\threshold} \nonumber \\ 
& \le 1 - \sum\limits_{i \in \nats[\threshold - 1]} \binom{\numProjections}{i} \fpr^{i} (1 - \fpr)^{i}
 \\
			& \cprob{}{\exists I \subseteq \nats[\numProjections], \setSize{I} \ge \numProjections - \threshold + 1: \forall \projIdx \in I, 
				\projFn[\projIdx](\genericMetricSpaceElmt) \neq \projFn[\projIdx](\genericMetricSpaceElmtAlt)}{\distanceNotation[\distance][\genericMetricSpaceElmt][\genericMetricSpaceElmtAlt] \le 
			\threshold} \nonumber \\ 
& \le 1 - \sum\limits_{i \in \nats[\numProjections - \threshold]} \binom{\numProjections}{i} \fnr^{i} (1 - \fnr)^{i}
 \\
		\end{align}
		\end{small}
where \fpr and \fnr are the false positive rate and false negative rate under each projection function.

\end{theorem}

\begin{proof}
The results follow from the fact that all of the projections are selected independently. Thus, the false positives and false negatives are binomially distributed with success probability \fpr and \fnr respectively. \eqref{eq:false_pos_bound} bounds the probability of getting at least $\threshold$ successes (false positives). 
Similarly, \eqref{eq:false_neg_bound} defines the probability of getting at least $\numProjections - \threshold + 1$ successes where a success is a false negative. 

\end{proof}

\section{Hiding correlations between queries}
\label{app:hiding_corr}
%


We will first make an assumption about the embedding function \embed, that is satisfied by most constructions of interest for content moderation~\cite{mckeown2023hamming,kulshrestha2021identifying}. The assumption roughly translates to the following: \embed embeds dissimilar items in the universe \universe to items in the metric space that are at least twice as far apart from each other, as the distance between the embeddings of similar items. 
More formally, 

\begin{assumption}
\label{assumption1}
The embedding function \embed embedding elements from \universe into metric space ensures that if two items $\genericUniverseElmt, \genericUniverseElmtAlt \in \universe$ are similar, then $\distanceNotation[\distance][\embed(\genericUniverseElmt) - \embed(\genericUniverseElmtAlt)] \le \threshold$. Otherwise, 
$\distanceNotation[\distance][\embed(\genericUniverseElmt) - \embed(\genericUniverseElmtAlt)] > \lshConst\threshold$ where $\lshConst > 2$ 
\end{assumption}

\begin{prop}
\label{thm:prob_match}
Let $\setNotation{A}, \setNotation{B} \subseteq \metricSpace$ be two sets such that $\setSize{A} = \setSize{B} = \blocklistSize$, and for every pair $\vec{a}, \vec{a'} \in \setNotation{A} \times \setNotation{A}$ (resp. $\vec{b}, \vec{b'} \in \setNotation{B} \times \setNotation{B}$), $\distanceNotation[\distance][\vec{a} - \vec{a'}] > t$ 
. Initialize \numProjections key-value stores $\projBuckets[1], \dots, \projBuckets[\numProjections]$. For each element $\vec{a} \in \setNotation{A}$, set $\projBuckets[\projIdx][\vec{a}] \assign \hashFn(\projFn[\projIdx](\vec{a}), \projFn[\projIdx])$ if some $\projBuckets[\projIdx][\vec{a'}] \assign \hashFn(\projFn[\projIdx](\vec{a}), \projFn[\projIdx])$, $\vec{a'} \in \setNotation{A}, \vec{a} \neq \vec{a'}$ does not already exist. Otherwise set $\projBuckets[\projIdx][\vec{a}] \assign \genericRand$ where $\genericRand \getsr \setDefn{0,1}^{\secParam}$. Create a fingerprint $\fingerprint{\vec{a}} \gets \setDefn{\projBuckets[\projIdx][\vec{a}]}_{\projIdx \in \nats[\numProjections]}$. Repeat the same process for the vectors in $\setNotation{B}$ after re-initializing the key-value stores. Then, 

\[
\cprob{}{\setSize{\fingerprint{\vec{a}} ~\cap \fingerprint{\vec{b}}} = \emptyset}{\distanceNotation[\distance][\vec{a} - \vec{b}] \le \threshold} \le (1 - \fnr + 2\blocklistSize \fpr)^{\numProjections}
\]
\end{prop}

\begin{proof}
Consider some projection function $\projFn[\projIdx]$. When $\distanceNotation[\distance][\vec{a} - \vec{b}] \le \threshold$, $\prob{\hashFn(\projFn[\projIdx](\vec{a}), \projFn[\projIdx])} \neq \prob{\hashFn(\projFn[\projIdx](\vec{a}), \projFn[\projIdx])} \le 1 - \fnr$. Also, $\hashFn(\projFn[\projIdx](\vec{a}), \projFn[\projIdx]) \notin \fingerprint{\vec{a}}$ if there exists some $\vec{a'} \in \setNotation{A}$ such that $\hashFn(\projFn[\projIdx](\vec{a}), \projFn[\projIdx]) = \hashFn(\projFn[\projIdx](\vec{a'}), \projFn[\projIdx])$ but $\distanceNotation[\distance][\vec{a} - \vec{a}] > \threshold$. This happens with probability at most $\fpr$. Summing over all vectors in $\setNotation{A}$, $\prob{\hashFn(\projFn[\projIdx](\vec{a}), \projFn[\projIdx]) \notin \fingerprint{\vec{a}}} \le \blocklistSize \fpr$. A similar argument holds for $\prob{\hashFn(\projFn[\projIdx](\vec{b}), \projFn[\projIdx]) \notin \fingerprint{\vec{b}}} \le \blocklistSize \fpr$. Thus, from a union bound, we have $\cprob{}{\setDefn{\projFn[\projIdx](\vec{a}) \cup \projFn[\projIdx](\vec{b})} \not\subseteq \fingerprint{\vec{b}} ~\cap \fingerprint{\vec{b}}}{\distanceNotation[\distance][\vec{a} - \vec{b}] \le \threshold} \le (1 - \fnr + 2\blocklistSize \fpr)$. The fact that the projections are independent gives the result. 
\end{proof}


\begin{theorem}
\label{thm:fnr}
Under Assumption~\ref{assumption1}, the strategy described to ensure the query invariant will ensure that the server will learn $\clientVec$ if there is some $\blocklistVec \in \blocklist$ such that $\hammingDist{\clientVec}{\blocklistVec} \le \threshold$ with probability at least $1 - (1 - \fnr + 2\blocklistSize \fpr)^{\numProjections}$ assuming that there are no false negatives in the cuckoo hash table. 
\end{theorem}




\begin{proof}
There are two conditions under which the input $\clientVec$'s projection may collide with the projection of an already uploaded content $\genericVec$ under the projection function $\projFn[\projIdx]$, which was not flagged by the server. The first case is where $\hammingDist{\clientVec}{\genericVec} > \threshold$. In this case, not querying on $\projFn[\projIdx](\clientVec)$ implies $\projFn[\projIdx](\genericVec) \in \fingerprint{\clientVec}$ and $\projFn[\projIdx](\clientVec) \notin \fingerprint{\clientVec}$. Due to \eqnref{eq:no_false_neg}, there is at least one function $\projFn[\projIdx'], \projIdx \neq \projIdx'$ such that $\projFn[\projIdx'](\clientVec) \neq \projFn[\projIdx'](\genericVec)$ with probability $1 - \fpr^{\numProjections}$. Also, due to \propref{thm:prob_match}, the probability that the server will find a matching element in the blocklist even when $\projFn[\projIdx](\clientVec) \notin \fingerprint{\clientVec}$ is not in the query is at least  $1 - (1 - \fnr + 2\blocklistSize \fpr)^{\numProjections}$. 

The second case is when $\hammingDist{\clientVec}{\genericVec} \le \threshold$. In this case, assume that $\blocklistVec$ is the vector (without the bit-mask) stored at the slot $\cuckooTable[\hashFn{1}(\prf{\prfKey}(\blindingScheme(\fingerprint{\blocklistVec}[\projIdx])))]$. Since, $\blocklistVec$ and $\genericVec$ did not match (as otherwise the server would have raised a flag), under Assumption~\ref{assumption1}, $\hammingDist{\blocklistVec}{\genericVec} \ge \lshConst \threshold$. Due to the triangle inequality, we have $\hammingDist{\blocklistVec}{\clientVec} + \hammingDist{\genericVec}{\clientVec} \ge \hammingDist{\genericVec}{\blocklistVec}$. From the above inequalities, this implies $\hammingDist{\blocklistVec}{\clientVec} \ge (\lshConst - 1) \cdot \threshold$. Now assuming that $\lshConst > 2$, $\hammingDist{\blocklistVec}{\clientVec} > \threshold$, which implies that the server will not learn $\clientVec$ despite the comparison $\blocklistVec$. A similar argument holds for the vector stored at $\cuckooTable[\hashFn{2}(\prf{\prfKey}(\blindingScheme(\fingerprint{\blocklistVec}[\projIdx])))]$. Thus, the projection can be dropped from the query without affecting the result. 
\end{proof}
