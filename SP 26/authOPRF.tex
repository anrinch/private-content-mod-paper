% We describe the protocol realizing \authOPRFFunc in \figref{fig:authOPRF}. The security of the protocol reduces to the secure of the 2HashDH OPRF construction~\cite{jarecki2014round} and the unforgeability of the randomizable set homomorphic scheme, proven in \thmref{thm:unforge}



\begin{algorithm}
\caption{\protocolAuthOPRF: protocol realizing \authOPRFFunc}
\label{protocol:authOPRF}
\begin{algorithmic}[1]



\Require A group \genGrp with order $\genericGrpOrd$ where the OM-gap Diffie Hellman problem is hard; 
a UF-CMA secure signature scheme \sigSchemeDefn; two random oracle $\oprfRO[1]: \universe \rightarrow \genGrp$ and $\oprfRO[2]: \bin^{\setSize{\universe} + \setSize{\genGrp}} \rightarrow \bin^{\secParam}$; collision-resistant hash function $\hashFn: \bin^{\secParam} \rightarrow \sigOutputSpace$ 

\Statex\Procedure{\publishList}{\blocklist}

 


\For{each $\auditor[\auditorIdx] \in \auditor[1], \dots, \auditor[\numAuditors]$}
\State Sample key pair $\langle \auditorPubKeyGenericScheme[\auditorIdx],\auditorPrivKeyGenericScheme[\auditorIdx] \rangle$ for \sigScheme
\State Sample $\blocklistSize$ random values $\genericRand[\auditorIdx, 1], \dots, \genericRand[\auditorIdx, \blocklistSize] \getsr \integersModArg{\genericGrpOrd}$ and broadcast the  values to all the other auditors.  

\State Wait until $\genericRand[j, 1], \dots, \genericRand [j, \blocklistSize]$ are received
from each auditor $\auditor[j], j \in [1, \numAuditors] \land~\auditorIdx \neq j$  

\For{each $\genericBlocklistElmt[\setIdx] \in \blocklist$}
\State Compute $\genericRand[\setIdx] \assign \prod\limits_{j \in [1, \numAuditors]} \genericRand[j, \setIdx]$, and sets $\blindedBlocklist = \{ \langle \setIdx, \oprfRO[1](\genericBlocklistElmt[\setIdx])^{\genericRand[\setIdx]} \rangle\}_{\setIdx \in \nats[\blocklistSize]}$ and $\genericSetVar \assign \{ \langle \setIdx, \genericBlocklistElmt[\setIdx], \genericRand[\setIdx] \rangle \}_{\setIdx \in \nats[\blocklistSize]}$
\EndFor
\State Sign $\auditorSig{\auditorIdx} \assign \signGenSig[\auditorPrivKeyGenericScheme[\auditorIdx]](\hashFn(\blindedBlocklist))$
\EndFor



\State The auditors publish \blindedBlocklist and $\{ \auditorSig{\auditorIdx} \}_{\auditorIdx \in [1, \numAuditors]}$, and provide \genericSetVar to \server

%Auditors invoke $\dssFunc$ to sign $\genericSig[\blocklist'] \assign \sign[\privKey](\hashFn(\blocklist'))$ and publish $\blocklist'$ and $\genericSig[\blocklist']$.

%\item Auditors invoke $\dssFunc$ to sign $\genericSig[\genericSetVar] \assign \sign[\privKey](\hashFn(\genericSetVar))$ and provide it to \server. 

%$\auditor[\auditorIdx]$ signs and provides $\sign[\auditorPrivKeyGenericScheme[\auditorIdx]](\hashFn(\genericSetVar))$ to \server 


%\item\label{authOPRF:auditor:sign_and_publish} $\auditor[\auditorIdx]$ signs and publishes $\genericSig[\blocklist']^{(\auditorIdx)} \assign \sign[\auditorPrivKeyGenericScheme[\auditorIdx]](\hashFn(\blocklist'))$ 

\EndProcedure

\Statex\Procedure{\evaluate}{\blindedBlocklist}


\State  After downloading $\blindedBlocklist$ and $\{ \auditorSig{\auditorPrivKeyGenericScheme[\auditorIdx]} \}_{\auditorIdx \in [1, \numAuditors]}$, \client aborts if $\verifyGenSig[\pubKey](\blindedBlocklist, \auditorPubKeyGenericScheme[\auditorIdx]) \neq 1$ for some $\auditorIdx \in [1, \numAuditors]$


\State \client initializes a set $\genericSetVar \assign \emptyset$ and samples a key $\prfKeyClient \sample \integersModArg{\genericGrpOrd}$

\State For each $\langle \setIdx, \oprfRO[1](\genericBlocklistElmt[\setIdx])^{\genericRand[\setIdx]} \rangle$ in $\blindedBlocklist$, \client  adds $\langle \setIdx,  (\oprfRO[1](\genericBlocklistElmt[\setIdx])^{\genericRand[\setIdx]})^{\prfKeyClient} \rangle$ to set $\genericSetVar'$. Send $\genericSetVar'$ to \server



\State\LComment{Upon receiving $\genericSetVar'$ from \client}
\State For every $\langle \setIdx,  (\oprfRO[1](\genericBlocklistElmt[\setIdx])^{\genericRand[\setIdx]})^{\prfKeyClient} \rangle \in \genericSetVar'$, let  $\langle \setIdx, \genericBlocklistElmt[\setIdx], \genericRand[\setIdx] \rangle \in \genericSetVar$. \server computes $((\oprfRO[1](\genericBlocklistElmt[\setIdx])^{\genericRand[\setIdx]})^{\prfKeyClient})^{\genericRand[\setIdx]^{-1}} = \oprfRO[1](\genericBlocklistElmt[\setIdx])^{\prfKeyClient}$ and outputs $\langle \genericBlocklistElmt[\setIdx], \oprfRO[2](\genericBlocklistElmt[\setIdx], \oprfRO[1](\genericBlocklistElmt[\setIdx])^{\prfKeyClient}) \rangle$. 
\EndProcedure
\end{algorithmic}
\end{algorithm}
