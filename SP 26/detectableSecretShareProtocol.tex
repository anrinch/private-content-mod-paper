
\begin{algorithm}
\caption{\detectableSecretSharingScheme: detectable secret sharing scheme\label{protocol:dss}}
\footnotesize
\begin{algorithmic}[1]

\Require A $(\secretShareLength, \secretShareThreshold)$ secret sharing scheme \secretSharingSchemeDefn; $(\secretShareLength - \secretShareThreshold, \secretShareThreshold - 1)$-detectable hash function $\dhf: \dhfKeySpace \times \dhfDomain \rightarrow \dhfRange$.

\Statex\LComment{Secret sharing a value $\genericSymmKey$ into $\secretShareLength$ shares}
\Procedure{\DSShare}{$\genericSymmKey$}   
\State Secret-share \genericSymmKey into \secretShareLength shares $\{ \secretShare{1}, \dots, \secretShare{\secretShareLength} \} \assign \SSShare(\genericSymmKey)$
%\State The dealer samples $\secretShareLength$ one-time pads, $\genericSymmKey_1, \dots, \genericSymmKey_{\secretShareLength}$ and computes the set $\ssSet \assign \{\langle 1,  \genericSymmKey_1, \secretShare{1}, \rangle, \dots, \langle \secretShareLength, \genericSymmKey_{\secretShareLength}, \secretShare{\secretShareLength} \rangle \}$
\State The dealer samples a key $\dhfKey \sample \dhfKeySpace$, and computes the set 
\[
\dhfSet \assign \{ \dhf[\dhfKey](\genericRand[\setIdx]): \genericRand[\setIdx] \sample \dhfDomain, \setIdx \in [1, \secretShareLength] \} 
\]

\State \Return $\ssSet \assign \{ \langle \secretShare{\setIdx}, \dhf[\dhfKey](\genericRand[\setIdx]) \rangle: \setIdx \in [1, \secretShareLength] \}$
\EndProcedure

\Statex\LComment{Reconstructing the detectable secret shares }
\Procedure{\DSCombine}{$\ssSet$} 
\LComment{Let $\ssSet = \{\langle \secretShare{\setIdx}', \dhfSetElmt{\setIdx}' \rangle: \setIdx \in [1,\secretShareLength] \}$ }
\State Compute the vector $\genericVec \assign \langle \dhfSetElmt{\setIdx}\rangle_{\setIdx \in [1, \secretShareLength]}$ 
\State Compute $\dhf[\dhfKey].\dhfDetect(\genericVec) =
    \matchedIndexSet$. If the algorithm fails, then abort. Otherwise, construct $\genericSetVar \assign \{ \secretShare{\setIdx}' :   \setIdx \in \matchedIndexSet \}$ and reconstruct the secret shares $\genericSymmKey \assign \SSCombine(\genericSetVar)$ 
\State \Return $\genericSymmKey$

\EndProcedure
\end{algorithmic}
\end{algorithm}
